# 数据储存

```c
int main()\
{
  char a = -128;
  // a:10000000
  //%u-打印无符号整数
  //发生整形提升
  //11111111111111111111111110000000
  printf("%u\n",a);
  //4,94,67,168
  rerurn 0;
}
```

```c
int main()
{
  char a =128;
  //由于128是整数，放在32位中，char只有八个比特位，最终截断，保留低符号位。
  //a存储在内存中：10000000
  //按照%u打印，对a进行整型提升
  //与上面的结果一样
  printf("%U",a);
  return 0;
}
```

```c
int main()
{
  int i = -20;
  unsigned int j =10;
  //1000000000000000000000000010100
  //1111111111111111111111111101011
  //1111111111111111111111111101100  ->-20的补码
  //0000000000000000000000000001010  ->10的补码
  //两者相加为：1111111111111111111111111110110
  //原码为：0000000000000000000000000001010
  printf("%d",i+j);
  return 0;
}
```

```c
int main()
{
  unsigned int i;
  for(i = 9;i>=0;i--)
  {
    printf("%u",i);
  }
  //死循环
  //9，8，7，6，5，4，3，2，1，0，很大的数，
}
```

```c
int main()
{
  char a[1000];
  int i;
  for(i=0:i<1000;i++)
  {
    a[i]=-1-i;
  }
  printf("%d",strlen(a));
  //strlen是求字符串的长度，找的\0的位置，统计\0之前出现多少个字符
  //'\0'的ASCII的码值为0
  //-129：
  //1000000000000000010000001
  //1111111111111111101111111
  //截断为127
  //最终为256
}
```
